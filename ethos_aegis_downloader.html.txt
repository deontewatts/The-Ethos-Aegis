async function buildAndDownload() {
  const btn = document.getElementById('dlBtn');
  const prog = document.getElementById('progress');
  const bar = document.getElementById('barFill');
  const status = document.getElementById('status');
  const done = document.getElementById('doneMsg');

  btn.disabled = true;
  btn.innerHTML = '<span>⏳</span> Building package…';
  prog.style.display = 'block';
  done.style.display = 'none';

  const zip = new JSZip();
  const entries = Object.entries(FILES);
  const total = entries.length;

  for (let i = 0; i < entries.length; i++) {
    const [path, content] = entries[i];
    const pct = Math.round(((i + 1) / total) * 90);
    bar.style.width = pct + '%';
    status.textContent = `Adding ${path}…`;
    await new Promise(r => setTimeout(r, 20));

    if (path.includes('/')) {
      const parts = path.split('/');
      const file = parts.pop();
      const folder = parts.join('/');
      zip.folder(folder).file(file, content);
    } else {
      zip.file(path, content);
    }
  }

  status.textContent = 'Compressing…';
  bar.style.width = '95%';
  await new Promise(r => setTimeout(r, 50));

  const blob = await zip.generateAsync({ type: 'blob', compression: 'DEFLATE' });
  bar.style.width = '100%';

  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'ethos-aegis.zip';
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  URL.revokeObjectURL(url);

  status.textContent = 'Download complete!';
  btn.disabled = false;
  btn.innerHTML = '<span>✓</span> Downloaded — click to re-download';
  done.style.display = 'block';
}
